253 
254  static BufferType to_buffer(const DataType &data,
255  const datatype::String &memory_type,
256  const dataspace::Dataspace &memory_space)
257  {
258  BufferType buffer = BufferType::create(memory_type,memory_space);
259 
260  auto iter = buffer.begin();
261  for(const auto &str: data)
262  {
263  std::copy(str.begin(),str.end(),iter);
264  std::advance(iter,memory_type.size()); //move iterator to the next position
265  }
266 
267  return buffer;
268  }
269 
270  static DataType from_buffer(const BufferType &buffer,
271  const datatype::String &memory_type,
272  const dataspace::Dataspace &)
273  {
274  DataType data;
275 
276  auto start=buffer.begin();
277  while(start!=buffer.end())
278  {
279  auto end = start+memory_type.size();
280  data.push_back(std::string(start,end));
281  start=end;
282  }
283  return data;
284  }
