672 template<typename T>
673 void Dataset::read(T &data,const dataspace::Selection &selection,
674  const property::DatasetTransferList &dtpl) const
675 {
676  auto memory_space = hdf5::dataspace::create(data);
677  auto memory_type = hdf5::datatype::create(data);
678 
679  dataspace::Dataspace file_space = dataspace();
680  file_space.selection(dataspace::SelectionOperation::SET,selection);
681  try{
682  const dataspace::Hyperslab & hyper = dynamic_cast<const dataspace::Hyperslab &>(selection);
683  auto dims = hyper.block();
684  auto count = hyper.count();
685  for(Dimensions::size_type i = 0; i != dims.size(); i++)
686  dims[i] *= count[i];
687 
688  dataspace::Simple selected_space(dims);
689  if (selected_space.size() == memory_space.size())
690  read(data,memory_type,selected_space,file_space,dtpl);
691  else
692  read(data,memory_type,memory_space,file_space,dtpl);
693  }
694  catch(const std::bad_cast&){
695  read(data,memory_type,memory_space,file_space,dtpl);
696  }
697 }
